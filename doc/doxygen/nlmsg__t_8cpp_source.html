56void nlmsg_t::put_attr(nlattr_t attr)
57{
58 int err
59 = std::visit([msgPtr=this->msgPtr_, name=attr.name](auto value) -> int
60 {
61 if constexpr(std::is_same_v<decltype(value), uint8_t>) {
62 return nla_put_u8(msgPtr, name, value);
63 }
64 else if constexpr(std::is_same_v<decltype(value), uint16_t>) {
65 return nla_put_u16(msgPtr, name, value);
66 }
67 else if constexpr(std::is_same_v<decltype(value), uint32_t>) {
68 return nla_put_u32(msgPtr, name, value);
69 }
70 else if constexpr(std::is_same_v<decltype(value), uint64_t>) {
71 return nla_put_u64(msgPtr, name, value);
72 }
73 else if constexpr(std::is_same_v<decltype(value), std::string>) {
74 return nla_put_string(msgPtr, name, value.data());
75 }
76 }, attr.value);
77
78 if(err < 0) {
79 throw std::runtime_error{nl_geterror(err)};
80 }
81}
