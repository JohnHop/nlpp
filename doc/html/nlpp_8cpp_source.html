118std::string nlpp::to_string(if_flags_t flags)
119{
120 std::string result;
121
122 for(int i = 0; i < flags.size()-1; ++i)
123 {
124 auto flag = (flags[i] ? 1 : 0) << i;
125 if(flag) {
126 result += flag ? ::to_string(static_cast<if_flag_e>(flag)) : "";
127 result += ",";
128 }
129 }
130
131 result.pop_back(); // remove last comma
132
133 return result;
134}

153if_index_t phy_lookup(std::string_view phy_name)
154{
155 auto const path
156 = std::filesystem::path{"/sys/class/ieee80211/"} / phy_name / "index";
157 if(!std::filesystem::exists(path)) {
158 throw std::runtime_error{std::format("device{} does not exists", phy_name)};
159 }
160
161 std::ifstream file{path};
162 if(!file.is_open()) {
163 throw std::runtime_error{std::format("Unable to open {}", path.string())};
164 }
165
166 std::array<char,200> buf{};
167 file.getline(buf.data(), 200, '\n');
168
169 return if_index_t{static_cast<uint32_t>(std::atoi(buf.data()))};
170}

173freq_chan_t nlpp::freq2chan(frequency_t freq) noexcept
174{
175 auto f = freq.get();
176 int res = 0;
177
178 if(f == 2484) {
179 res = 14;
180 } else if(f < 2484) {
181 res = (f - 2407) / 5;
182 } else if(f < 45000) { /* FIXME: dot11ChannelStartingFactor (802.11-2007 17.3.8.3.2) */
183 res = f/5 - 1000;
184 } else if(f >= 58320 && f <= 64800) {
185 res = (f - 56160) / 2160;
186 }
187 //? else throw std::invalid_argument
188
189 return freq_chan_t{res};
190}

193frequency_t nlpp::chan2freq(freq_chan_t chan) noexcept
194{
195 auto c = chan.get();
196 frequency_t::value_type res = (c + 1000) * 5;
197
198 if(c < 14) {
199 res = 2407 + c * 5;

201 res = 2484;

206 return frequency_t{res};
