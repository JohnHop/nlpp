47[[nodiscard]]
48dev_info_t NetlinkGeneric::get_interface(if_index_t ifindex)
49{
50 std::map<uint32_t,dev_info_t> interface_info;
51
52 if(ifindex.get() == 0) {
53 throw std::invalid_argument{"interface index cannot be zero"};
54 }
55
56 nlmsg_t msg{this->nl80211_id, nl80211_commands::NL80211_CMD_GET_INTERFACE};
57 msg.put_attr({nl80211_attrs::NL80211_ATTR_IFINDEX, ifindex.get()});
58
59 this->send_msg(
60 std::move(msg), NetlinkGeneric::get_interface_handler, &interface_info);
61
62 return interface_info.at(ifindex.get());
63}

132dev_capability_t NetlinkGeneric::get_phy(if_index_t phy_index)
133{
134 dev_capability_t result;
135
136 // first, send `NL80211_CMD_GET_PROTOCOL_FEATURES`
137 bool nl80211_has_split_wiphy{};
138
139 nlmsg_t msg{this->nl80211_id,
140 nl80211_commands::NL80211_CMD_GET_PROTOCOL_FEATURES};
141
142 this->send_msg(std::move(msg),
143 &NetlinkGeneric::get_feature_handler, &nl80211_has_split_wiphy);
144
145 // second, send `NL80211_CMD_GET_WIPHY` with appropriate flags
146 msg = nlmsg_t{this->nl80211_id, nl80211_commands::NL80211_CMD_GET_WIPHY};
147
148 msg.put_attr(
149 {NL80211_ATTR_WIPHY, static_cast<uint32_t>(phy_index.get())}
150 );
151
152 if(nl80211_has_split_wiphy) {
153 msg.put_flag(NL80211_ATTR_SPLIT_WIPHY_DUMP);
154 msg.nlmsg_hdr()->nlmsg_flags |= NLM_F_DUMP;
155 }
156
157 this->send_msg(std::move(msg),
158 &NetlinkGeneric::get_phy_handler, &result);
159
160 return result;
161}

185 socket_.send_auto(msg);

209int NetlinkGeneric::get_interface_handler(struct nl_msg* msg, void* arg) noexcept
210{
211 dev_info_t dev_info;
212
213 auto* resultPtr = reinterpret_cast<std::map<uint32_t,dev_info_t>*>(arg);
214
215 auto gnlh = reinterpret_cast<struct genlmsghdr*>(nlmsg_data(nlmsg_hdr(msg)));
216 struct nlattr* tb_msg[NL80211_ATTR_MAX + 1];
217 uint32_t ifindex = 0;
218
219 nla_parse(tb_msg, NL80211_ATTR_MAX, genlmsg_attrdata(gnlh, 0),
220 genlmsg_attrlen(gnlh, 0), nullptr);
221
222 if(tb_msg[NL80211_ATTR_IFNAME]) {
223 dev_info.if_name = std::string(
224 nla_get_string(tb_msg[NL80211_ATTR_IFNAME]));
225 }
226 if(tb_msg[NL80211_ATTR_IFINDEX]) {
227 dev_info.if_index.get()
228 = ifindex = nla_get_u32(tb_msg[NL80211_ATTR_IFINDEX]);
229 }
230 if(tb_msg[NL80211_ATTR_WDEV]) {
231 dev_info.wdev = nla_get_u64(tb_msg[NL80211_ATTR_WDEV]);
232 }
233 // if(tb_msg[NL80211_ATTR_MAC]) { }
234 // if(tb_msg[NL80211_ATTR_SSID]) { }
235 if(tb_msg[NL80211_ATTR_IFTYPE]) {
236 dev_info.type =
237 static_cast<if_type_e>(nla_get_u32(tb_msg[NL80211_ATTR_IFTYPE]));
238 }
239 if(tb_msg[NL80211_ATTR_WIPHY]) {
240 dev_info.wiphy_index.get() = nla_get_u32(tb_msg[NL80211_ATTR_WIPHY]);
241 }
242 if(tb_msg[NL80211_ATTR_WIPHY_FREQ])
243 {
244 dev_info.wiphy_freq.get() = nla_get_u32(tb_msg[NL80211_ATTR_WIPHY_FREQ]);
245
246 if(tb_msg[NL80211_ATTR_CHANNEL_WIDTH]) {
247 dev_info.channel_width =
248 nla_get_u32(tb_msg[NL80211_ATTR_CHANNEL_WIDTH]);
249 }
250 }
251
252 // for now, we insert the device info inside `resultPtr` map only if it is not
253 // a "Unnamed/non-netdev interface"
254 if(ifindex) {
255 resultPtr->insert({ifindex,std::move(dev_info)});
256 }
257
258 return NL_SKIP;
259}

263int NetlinkGeneric::get_feature_handler(struct nl_msg *msg, void *arg) noexcept
264{
265 bool* nl80211_has_split_wiphy = reinterpret_cast<bool*>(arg);
266
267 struct nlattr* tb_msg[NL80211_ATTR_MAX + 1];
268 auto gnlh = reinterpret_cast<struct genlmsghdr*>(nlmsg_data(nlmsg_hdr(msg)));
269
270 nla_parse(tb_msg, NL80211_ATTR_MAX, genlmsg_attrdata(gnlh, 0),
271 genlmsg_attrlen(gnlh, 0), nullptr);
272
273 if(tb_msg[NL80211_ATTR_PROTOCOL_FEATURES])
274 {
275 uint32_t feat = nla_get_u32(tb_msg[NL80211_ATTR_PROTOCOL_FEATURES]);
276
277 if(feat & NL80211_PROTOCOL_FEATURE_SPLIT_WIPHY_DUMP) {
278 *nl80211_has_split_wiphy = true;
279 }
280 }
281
282 return NL_SKIP;
283}

287int NetlinkGeneric::get_phy_handler(struct nl_msg* msg, void* arg) noexcept
288{
289 struct nlattr* tb_msg[NL80211_ATTR_MAX + 1];
290 auto gnlh = reinterpret_cast<struct genlmsghdr *>(nlmsg_data(nlmsg_hdr(msg)));
291 struct nlattr* tb_band[NL80211_BAND_ATTR_MAX + 1];
292 struct nlattr* tb_freq[NL80211_FREQUENCY_ATTR_MAX + 1];
293
294 static struct nla_policy freq_policy[NL80211_FREQUENCY_ATTR_MAX + 1]{};
295 freq_policy[NL80211_FREQUENCY_ATTR_FREQ] = { .type = NLA_U32 };
296 freq_policy[NL80211_FREQUENCY_ATTR_DISABLED] = { .type = NLA_FLAG };
297 freq_policy[NL80211_FREQUENCY_ATTR_PASSIVE_SCAN] = { .type = NLA_FLAG };
298 freq_policy[NL80211_FREQUENCY_ATTR_NO_IBSS] = { .type = NLA_FLAG };
299 freq_policy[NL80211_FREQUENCY_ATTR_RADAR] = { .type = NLA_FLAG };
300 freq_policy[NL80211_FREQUENCY_ATTR_MAX_TX_POWER] = { .type = NLA_U32 };
301
302 static int last_band = -1;
303 static uint32_t phy_id = -1;
304 static bool band_had_freq = false;
305
306 // Get the return value pointer
307 auto capabilityPtr = reinterpret_cast<dev_capability_t* const>(arg);
308
309 nla_parse(tb_msg, NL80211_ATTR_MAX, genlmsg_attrdata(gnlh, 0),
310 genlmsg_attrlen(gnlh, 0), nullptr);
311
312 if(tb_msg[NL80211_ATTR_WIPHY])
313 {
314 if(nla_get_u32(tb_msg[NL80211_ATTR_WIPHY]) != phy_id) {
315 last_band = -1;
316 }
317 capabilityPtr->wiphy_index.get()
318 = phy_id = nla_get_u32(tb_msg[NL80211_ATTR_WIPHY]);
319 // FIXME: probabilmente dobbiamo rrestituire una mappa di capability_info_t
320 }
321
322 if(tb_msg[NL80211_ATTR_WIPHY_NAME]) {
323 capabilityPtr->wiphy_name = nla_get_string(tb_msg[NL80211_ATTR_WIPHY_NAME]);
324 }
325
326 if(tb_msg[NL80211_ATTR_SUPPORTED_IFTYPES])
327 {
328 struct nlattr* nl_mode;
329 int rem_mode;
330 nla_for_each_nested(nl_mode, tb_msg[NL80211_ATTR_SUPPORTED_IFTYPES], rem_mode) {
331 capabilityPtr->iftypes.emplace_back(static_cast<if_type_e>(nla_type(nl_mode)));
332 }
333 }
334
335 if(tb_msg[NL80211_ATTR_WIPHY_BANDS])
336 {
337 struct nlattr* nl_band;
338 int rem_band;
339
340 nla_for_each_nested(nl_band, tb_msg[NL80211_ATTR_WIPHY_BANDS], rem_band)
341 {
342 if(last_band != nl_band->nla_type) {
343 band_had_freq = false;
344 }
345 last_band = nl_band->nla_type;
346
347 nla_parse(tb_band, NL80211_BAND_ATTR_MAX,
348 reinterpret_cast<struct nlattr*>(nla_data(nl_band)), nla_len(nl_band),
349 nullptr);
350
351 if(tb_band[NL80211_BAND_ATTR_FREQS])
352 {
353 if(!band_had_freq) {
354 band_had_freq = true;
355 }
356
357 struct nlattr *nl_freq;
358 int rem_freq;
359 nla_for_each_nested(nl_freq, tb_band[NL80211_BAND_ATTR_FREQS], rem_freq)
360 {
361 nla_parse(tb_freq, NL80211_FREQUENCY_ATTR_MAX,
362 reinterpret_cast<struct nlattr*>(nla_data(nl_freq)),
363 nla_len(nl_freq), freq_policy);
364
365 if(!tb_freq[NL80211_FREQUENCY_ATTR_FREQ]) {
366 continue;
367 }
368
369 capabilityPtr->freqs.emplace_back(
370 nla_get_u32(tb_freq[NL80211_FREQUENCY_ATTR_FREQ]) );
371 }
372 }
373 }
374 }
375
376 if(tb_msg[NL80211_ATTR_SUPPORTED_COMMANDS])
377 {
378 struct nlattr *nl_cmd;
379 int rem_cmd;
380 nla_for_each_nested(nl_cmd, tb_msg[NL80211_ATTR_SUPPORTED_COMMANDS], rem_cmd)
381 {
382 capabilityPtr->cmds.push_back(
383 static_cast<nl80211_commands>(nla_get_u32(nl_cmd)) );
384 }
385 }
386
387 return NL_SKIP;
388}
